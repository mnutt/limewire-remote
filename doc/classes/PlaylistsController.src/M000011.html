  def index
    @playlists = Playlist.find(:all, :order => "list_position ASC")
    @ordered_playlists = []
    @playlists.each_with_index do |p, i| 
      @ordered_playlists << { :playlist => p.to_playlist, :position => i, :is_owner => true } 
    end

    render :json => @ordered_playlists.to_json()
  end
